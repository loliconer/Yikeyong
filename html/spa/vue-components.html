<vue-table :source="buildings" :columns="buildingColumns" :default-columns="[]" :count-of-page="5" simple checkbox fixed-head @check="getSelectedBuildings" @click-row="clickRow" keep-sort>
   <template slot="age" scope="scope">
      <div>修改列的默认显示：{ {scope.value.age} }</div>
   </template>
   <template slot="operation" scope="scope">
       <button class="btn" type="button">{ {scope.value.name} }</button>
   </template>
</vue-table>
//columns: 指定显示的列，列可排序时，sortable属性为true，自定义的列的custom为true
//default-columns: 默认显示的列，不传则与columns相同
//count-of-page: 每页显示的条数
//simple: 简化版表格
//checkbox: 显示可勾选的列
//fixed-head: 固定表头，只在simple模式下有效
//@check: 数组，获取勾选的记录，全选表示选中当前页的所有条数
//@click-row: 行点击事件，可以获取点击行的对象
//keep-sort：当数据源变化时保持排序状态不变

树形列表
<ul>
    <vue-tree-list :source="items" @select="selectItem" @unselect="unselectItem" :trigger="trigger"></vue-tree-list>
</ul>
//source: 数据源
//select: 勾选节点时触发，获取勾选的项
//unselect: 取消勾选时触发，获取取消勾选的项
//trigger: 更新整棵树的状态
new Vue({
  data: {
    trigger: false,
    items: {
      name: 'IT',
      children: [
        {
          name: 'Front End',
          children: [
            {
              name: 'Lib',
              children: [
                {
                  name: 'Vue'
                },
                {
                  name: 'React'
                },
                {
                  name: 'Angular'
                }
              ]
            }
          ]
        },
        {
          name: 'Back End'
        }
      ]
    }
  },
  methods: {
    selectItem(items) {
      console.log(items)
      this.trigger = !this.trigger
    },
    unselectItem(items) {
      console.log(items)
      this.trigger = !this.trigger
    }
  }
})
